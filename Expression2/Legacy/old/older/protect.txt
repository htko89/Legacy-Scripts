@name Protect
@inputs [Attacker]:entity [Damage]:number
@outputs [Entity]:entity [On Retaliate]:number
@persist

Entity=owner()
On=1
if(Damage){
    Retaliate=1
    timer("Retaliate",Damage*50)
}
if(Retaliate){
    Bomb=propSpawn("models/props_junk/gascan001a.mdl", Attacker:pos(), 0)
    Bomb:propBreak()
}
if(clk("Retaliate")){
    Retaliate=0
}
