@name Car
@inputs [W A S D]:number
@outputs [Length Power]:number
@persist [LR FB MLength TLength TPower]:number
@trigger all

runOnTick(1)

MLength=10
TLength=64.27
TPower=10000

if(W){LR=1}else{LR=0}
if(S){LR=-1}else{LR=0}
LR*=MLength
if(A){FB=1}else{LR=0}
if(D){FB=-1}else{LR=0}

Power=FB*TPower
Length=LR+TLength
